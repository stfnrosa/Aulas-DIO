box-sizing: content-box; tamanho = largura+padding+border 
Exemplos:
        objeto1{
        margin:20px;
        boder:4px;
        padding:10px;
        height:300px;
        width:300px;
        box-sizing:content-box;
        }
        tamanho = largura 300px + 
        padding 10 left e 10 right +
        boder 4 left e 4 right = 328px;
box-sizing: border-box; largura = tamanho-padding-border 
Exemplos:
        objeto2{
        margin:20px;
        boder:4px;
        padding:10px;
        height:300px;
        width:300px;
        box-sizing:border-box;
        }
        largura = tamanho 300px - 
        padding 10 left e 10 right -
        boder 4 left e 4 right = 272px;
